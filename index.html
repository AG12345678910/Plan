display: block; margin: 0 auto; } const canvas = document.getElementById("gameCanvas"); const ctx = canvas.getContext("2d"); // Set canvas size canvas.width = 800; canvas.height = 400; // Player settings let player = { x: 50, y: canvas.height / 2, size: 30, speed: 3, dx: 0, dy: 0 }; // Obstacle settings let obstacles = []; let obstacleSpeed = 2; // Create player function drawPlayer() { ctx.fillStyle = "yellow"; ctx.fillRect(player.x, player.y, player.size, player.size); } // Move player function movePlayer() { player.x += player.dx; player.y += player.dy; } // Create obstacles function createObstacles() { if (Math.random() < 0.02) { let obstacleY = Math.random() * (canvas.height - 30); obstacles.push({ x: canvas.width, y: obstacleY, width: 30, height: 30 }); } } // Move obstacles function moveObstacles() { for (let i = 0; i < obstacles.length; i++) { obstacles[i].x -= obstacleSpeed; if (obstacles[i].x < 0) { obstacles.splice(i, 1); } } } // Draw obstacles function drawObstacles() { for (let i = 0; i < obstacles.length; i++) { ctx.fillStyle = "red"; ctx.fillRect(obstacles[i].x, obstacles[i].y, obstacles[i].width, obstacles[i].height); } } // Collision detection function detectCollisions() { for (let i = 0; i < obstacles.length; i++) { if ( player.x < obstacles[i].x + obstacles[i].width && player.x + player.size > obstacles[i].x && player.y < obstacles[i].y + obstacles[i].height && player.y + player.size > obstacles[i].y ) { alert("Game Over!"); resetGame(); } } } // Keyboard controls document.addEventListener("keydown", (e) => { if (e.key === "ArrowUp") player.dy = -player.speed; if (e.key === "ArrowDown") player.dy = player.speed; if (e.key === "ArrowLeft") player.dx = -player.speed; if (e.key === "ArrowRight") player.dx = player.speed; }); document.addEventListener("keyup", (e) => { if (e.key === "ArrowUp" || e.key === "ArrowDown") player.dy = 0; if (e.key === "ArrowLeft" || e.key === "ArrowRight") player.dx = 0; }); // Reset the game function resetGame() { player.x = 50; player.y = canvas.height / 2; obstacles = []; } // Update game elements function update() { ctx.clearRect(0, 0, canvas.width, canvas.height); movePlayer(); createObstacles(); moveObstacles(); drawPlayer(); drawObstacles(); detectCollisions(); } // Game loop setInterval(update, 1000 / 60);
