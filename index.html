border: 2px solid white; } const canvas = document.getElementById('pongCanvas'); const ctx = canvas.getContext('2d'); // Set canvas dimensions canvas.width = 800; canvas.height = 400; // Create the pong paddle const paddleWidth = 10, paddleHeight = 100; let paddle1Y = (canvas.height - paddleHeight) / 2, paddle2Y = (canvas.height - paddleHeight) / 2; let paddleSpeed = 5; // Create the pong ball const ballSize = 10; let ballX = canvas.width / 2, ballY = canvas.height / 2, ballSpeedX = 5, ballSpeedY = 5; // Draw the paddles function drawPaddle(x, y) { ctx.fillStyle = "#fff"; ctx.fillRect(x, y, paddleWidth, paddleHeight); } // Draw the ball function drawBall() { ctx.fillStyle = "#fff"; ctx.fillRect(ballX, ballY, ballSize, ballSize); } // Move the paddles function movePaddle() { if (upPressed && paddle1Y > 0) { paddle1Y -= paddleSpeed; } if (downPressed && paddle1Y + paddleHeight < canvas.height) { paddle1Y += paddleSpeed; } if (wPressed && paddle2Y > 0) { paddle2Y -= paddleSpeed; } if (sPressed && paddle2Y + paddleHeight < canvas.height) { paddle2Y += paddleSpeed; } } // Move the ball function moveBall() { ballX += ballSpeedX; ballY += ballSpeedY; // Ball collision with top/bottom walls if (ballY <= 0 || ballY + ballSize >= canvas.height) { ballSpeedY = -ballSpeedY; } // Ball collision with paddles if (ballX <= paddleWidth && ballY >= paddle1Y && ballY <= paddle1Y + paddleHeight || ballX + ballSize >= canvas.width - paddleWidth && ballY >= paddle2Y && ballY <= paddle2Y + paddleHeight) { ballSpeedX = -ballSpeedX; } // Ball out of bounds (reset) if (ballX <= 0 || ballX + ballSize >= canvas.width) { ballX = canvas.width / 2; ballY = canvas.height / 2; ballSpeedX = -ballSpeedX; } } // Keyboard controls let upPressed = false, downPressed = false, wPressed = false, sPressed = false; document.addEventListener('keydown', (e) => { if (e.key === 'ArrowUp') upPressed = true; if (e.key === 'ArrowDown') downPressed = true; if (e.key === 'w') wPressed = true; if (e.key === 's') sPressed = true; }); document.addEventListener('keyup', (e) => { if (e.key === 'ArrowUp') upPressed = false; if (e.key === 'ArrowDown') downPressed = false; if (e.key === 'w') wPressed = false; if (e.key === 's') sPressed = false; }); // Update the game function update() { movePaddle(); moveBall(); draw(); } // Draw everything function draw() { ctx.clearRect(0, 0, canvas.width, canvas.height); drawPaddle(0, paddle1Y); drawPaddle(canvas.width - paddleWidth, paddle2Y); drawBall(); } // Game loop setInterval(update, 1000 / 60);
